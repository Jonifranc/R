** =reshape=: =wide= a =long=
- Añadimos argumentos
#+begin_src R 
  CO2long <- reshape(CO2,
                     varying=list(names(CO2)[5:16]),
                     timevar='Year', v.names='Value',
                     times=2000:2011,
                     direction='long')
  head(CO2long)
#+end_src
** =reshape=: =long= a =wide=
