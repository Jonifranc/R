\begin{frame}[fragile]

\label{sec-3-2-4}


\lstset{language=R}
\begin{lstlisting}
halfHour <- function(tt, delta=30){
  tt <- as.POSIXlt(tt)
  gg <- tt$min %/% delta
  tt <- modifyList(tt, list(min=gg*delta))
  as.POSIXct(tt)
}
\end{lstlisting}


\lstset{language=R}
\begin{lstlisting}
hawaii30 <- aggregate(hawaii, by=halfHour, FUN=sum)/60
